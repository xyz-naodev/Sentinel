EXAMPLE USAGE
=============

// Initialize entire system
SentinelCore.init({
  mapContainer: 'map',
  enableMapTracking: true,
  enableIncidentSimulation: true,
  autoInitMap: true
});

// Get all units
const units = UnitTracker.getAllUnits();
// Output: [
//   { id: 'T01', name: 'Unit 01', position: {...}, status: 'on-duty' },
//   { id: 'T02', name: 'Unit 02', position: {...}, status: 'patrolling' },
//   { id: 'T03', name: 'Unit 03', position: {...}, status: 'idle' }
// ]

// Get incident stats
const stats = IncidentManager.getStats();
// Output: { total: 15, active: 3, resolved: 12, typeDistribution: {...} }

// Create incident
IncidentManager.createIncident({
  type: 'Crime',
  location: 'Park',
  lat: 14.345,
  lng: 121.115
});
// Automatically: adds to map, shows notification, updates badge

// Get active units
const active = UnitTracker.getUnitsByStatus('on-duty');

// Show notification
UIUtils.showSuccess('Success', 'Incident resolved');

// Render lists
UIUtils.renderUnitList(units, 'unitsList');
UIUtils.renderIncidentList(IncidentManager.getRecent(5), 'incidentsList');
